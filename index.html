# Endpoints

# Auth

  • login : POST /auth/login
  • me : GET /auth/me

# Deelnemers

  • activate : POST /deelnemers/activate
  • all : GET /deelnemers/all
  • create : POST /deelnemers/create
  • id : GET /deelnemers/:id
  • id : PATCH /deelnemers/:id
  • verify-code : POST /deelnemers/verify-code

# Icons

  • all : GET /icons/all
  • create : POST /icons/create
  • id : GET /icons/:id
  • remove : DELETE /icons/:id/remove
